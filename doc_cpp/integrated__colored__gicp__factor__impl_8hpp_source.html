95template <typename TargetFrame, typename SourceFrame, typename IntensityGradients>
96void IntegratedColoredGICPFactor_<TargetFrame, SourceFrame, IntensityGradients>::update_correspondences(const Eigen::Isometry3d& delta) const {
97 bool do_update = true;
98 if (correspondences.size() == frame::size(*source) && (correspondence_update_tolerance_trans > 0.0 || correspondence_update_tolerance_rot > 0.0)) {
99 Eigen::Isometry3d diff = delta.inverse() * last_correspondence_point;
100 double diff_rot = Eigen::AngleAxisd(diff.linear()).angle();
101 double diff_trans = diff.translation().norm();
102 if (diff_rot < correspondence_update_tolerance_rot && diff_trans < correspondence_update_tolerance_trans) {
103 do_update = false;
104 }
105 }
106
107 if (do_update) {
108 last_correspondence_point = delta;
109 }
110
111 correspondences.resize(frame::size(*source));
112 mahalanobis.resize(frame::size(*source));
113
114 const auto perpoint_task = [&](int i) {
115 if (do_update) {
116 Eigen::Vector4d pt = delta * frame::point(*source, i);
117 pt[3] = frame::intensity(*source, i);
118
119 size_t k_index = -1;
120 double k_sq_dist = -1;
121
122 size_t num_found = target_tree->knn_search(pt.data(), 1, &k_index, &k_sq_dist, max_correspondence_distance_sq);
123 correspondences[i] = (num_found && k_sq_dist < max_correspondence_distance_sq) ? k_index : -1;
124 }
125
126 if (correspondences[i] < 0) {
127 mahalanobis[i].setIdentity();
128 } else {
129 const auto& target_cov = frame::cov(*target, correspondences[i]);
130 Eigen::Matrix4d RCR = (target_cov + delta.matrix() * frame::cov(*source, i) * delta.matrix().transpose());
131 RCR(3, 3) = 1.0;
132 mahalanobis[i] = RCR.inverse();
133 mahalanobis[i](3, 3) = 0.0;
134 }
135 };
136
137 if (is_omp_default() || num_threads == 1) {
138#pragma omp parallel for num_threads(num_threads) schedule(guided, 8)
139 for (int i = 0; i < frame::size(*source); i++) {
140 perpoint_task(i);
141 }
142 } else {
143#ifdef GTSAM_POINTS_USE_TBB
144 tbb::parallel_for(tbb::blocked_range<int>(0, frame::size(*source), 8), [&](const tbb::blocked_range<int>& range) {
145 for (int i = range.begin(); i < range.end(); i++) {
146 perpoint_task(i);
147 }
148 });
149#else
150 std::cerr << "error: TBB is not available" << std::endl;
151 abort();
152#endif
153 }
154}

156template <typename TargetFrame, typename SourceFrame, typename IntensityGradients>
157double IntegratedColoredGICPFactor_<TargetFrame, SourceFrame, IntensityGradients>::evaluate(
158 const Eigen::Isometry3d& delta,
159 Eigen::Matrix<double, 6, 6>* H_target,
160 Eigen::Matrix<double, 6, 6>* H_source,
161 Eigen::Matrix<double, 6, 6>* H_target_source,
162 Eigen::Matrix<double, 6, 1>* b_target,
163 Eigen::Matrix<double, 6, 1>* b_source) const {
164 //
165 if (correspondences.size() != frame::size(*source)) {
166 update_correspondences(delta);
167 }
168
169 const double geometric_term_weight = 1.0 - photometric_term_weight;
170
171 const auto perpoint_task = [&](
172 int i,
173 Eigen::Matrix<double, 6, 6>* H_target,
174 Eigen::Matrix<double, 6, 6>* H_source,
175 Eigen::Matrix<double, 6, 6>* H_target_source,
176 Eigen::Matrix<double, 6, 1>* b_target,
177 Eigen::Matrix<double, 6, 1>* b_source) {
178 const long target_index = correspondences[i];
179 if (target_index < 0) {
180 return 0.0;
181 }
182
183 // source atributes
184 const auto& mean_A = frame::point(*source, i);
185 const auto& cov_A = frame::cov(*source, i);
186 const double intensity_A = frame::intensity(*source, i);
187
188 // target attributes
189 const auto& mean_B = frame::point(*target, target_index);
190 const auto& cov_B = frame::cov(*target, target_index);
191 const auto& normal_B = frame::normal(*target, target_index);
192 const auto& gradient_B = frame::intensity_gradient(*target_gradients, target_index);
193 const double intensity_B = frame::intensity(*target, target_index);
194
195 const Eigen::Vector4d transed_A = delta * mean_A;
196
197 // geometric error
198 const Eigen::Vector4d residual_geom = transed_A - mean_B;
199 const double error_geom = residual_geom.transpose() * geometric_term_weight * mahalanobis[i] * residual_geom;
200
201 // photometric error
202 const Eigen::Vector4d projected = transed_A - (transed_A - mean_B).dot(normal_B) * normal_B;
203 const Eigen::Vector4d offset = projected - mean_B;
204 const double residual_photo = intensity_B + gradient_B.dot(offset) - intensity_A;
205
206 const double error_photo = residual_photo * photometric_term_weight * residual_photo;
207
208 if (!H_target) {
209 return error_geom + error_photo;
210 }
211
212 Eigen::Matrix<double, 4, 6> J_transed_target = Eigen::Matrix<double, 4, 6>::Zero();
213 J_transed_target.block<3, 3>(0, 0) = gtsam::SO3::Hat(transed_A.head<3>());
214 J_transed_target.block<3, 3>(0, 3) = -Eigen::Matrix3d::Identity();
215
216 Eigen::Matrix<double, 4, 6> J_transed_source = Eigen::Matrix<double, 4, 6>::Zero();
217 J_transed_source.block<3, 3>(0, 0) = -delta.linear() * gtsam::SO3::Hat(mean_A.template head<3>());
218 J_transed_source.block<3, 3>(0, 3) = delta.linear();
219
220 // geometric error derivatives
221 const auto& J_egeom_target = J_transed_target;
222 const auto& J_egeom_source = J_transed_source;
223
224 *H_target += J_egeom_target.transpose() * geometric_term_weight * mahalanobis[i] * J_egeom_target;
225 *H_source += J_egeom_source.transpose() * geometric_term_weight * mahalanobis[i] * J_egeom_source;
226 *H_target_source += J_egeom_target.transpose() * geometric_term_weight * mahalanobis[i] * J_egeom_source;
227 *b_target += J_egeom_target.transpose() * geometric_term_weight * mahalanobis[i] * residual_geom;
228 *b_source += J_egeom_source.transpose() * geometric_term_weight * mahalanobis[i] * residual_geom;
229
230 // photometric error derivatives
231 Eigen::Matrix<double, 4, 4> J_projected_transed = Eigen::Matrix4d::Identity() - normal_B * normal_B.transpose();
232 J_projected_transed(3, 3) = 0.0;
233 const auto& J_offset_transed = J_projected_transed;
234
235 Eigen::Matrix<double, 1, 4> J_ephoto_offset = gradient_B;
236 Eigen::Matrix<double, 1, 4> J_ephoto_transed = J_ephoto_offset * J_offset_transed;
237
238 Eigen::Matrix<double, 1, 6> J_ephoto_target = J_ephoto_transed * J_transed_target;
239 Eigen::Matrix<double, 1, 6> J_ephoto_source = J_ephoto_transed * J_transed_source;
240
241 *H_target += J_ephoto_target.transpose() * photometric_term_weight * J_ephoto_target;
242 *H_source += J_ephoto_source.transpose() * photometric_term_weight * J_ephoto_source;
243 *H_target_source += J_ephoto_target.transpose() * photometric_term_weight * J_ephoto_source;
244 *b_target += J_ephoto_target.transpose() * photometric_term_weight * residual_photo;
245 *b_source += J_ephoto_source.transpose() * photometric_term_weight * residual_photo;
246
247 return error_geom + error_photo;
248 };
249
250 if (is_omp_default() || num_threads == 1) {
251 return scan_matching_reduce_omp(perpoint_task, frame::size(*source), num_threads, H_target, H_source, H_target_source, b_target, b_source);
252 } else {
253 return scan_matching_reduce_tbb(perpoint_task, frame::size(*source), H_target, H_source, H_target_source, b_target, b_source);
254 }
255}
