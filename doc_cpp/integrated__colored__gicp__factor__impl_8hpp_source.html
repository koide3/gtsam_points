100template <typename TargetFrame, typename SourceFrame, typename IntensityGradients>
101void IntegratedColoredGICPFactor_<TargetFrame, SourceFrame, IntensityGradients>::update_correspondences(const Eigen::Isometry3d& delta) const {
102 bool do_update = true;
103 if (correspondences.size() == frame::size(*source) && (correspondence_update_tolerance_trans > 0.0 || correspondence_update_tolerance_rot > 0.0)) {
104 Eigen::Isometry3d diff = delta.inverse() * last_correspondence_point;
105 double diff_rot = Eigen::AngleAxisd(diff.linear()).angle();
106 double diff_trans = diff.translation().norm();
107 if (diff_rot < correspondence_update_tolerance_rot && diff_trans < correspondence_update_tolerance_trans) {
108 do_update = false;
109 }
110 }
111
112 if (do_update) {
113 last_correspondence_point = delta;
114 }
115
116 correspondences.resize(frame::size(*source));
117 mahalanobis.resize(frame::size(*source));
118
119 const auto perpoint_task = [&](int i) {
120 if (do_update) {
121 Eigen::Vector4d pt = delta * frame::point(*source, i);
122 pt[3] = frame::intensity(*source, i);
123
124 size_t k_index = -1;
125 double k_sq_dist = -1;
126
127 size_t num_found = target_tree->knn_search(pt.data(), 1, &k_index, &k_sq_dist, max_correspondence_distance_sq);
128 correspondences[i] = (num_found && k_sq_dist < max_correspondence_distance_sq) ? k_index : -1;
129 }
130
131 if (correspondences[i] < 0) {
132 mahalanobis[i].setIdentity();
133 } else {
134 const auto& target_cov = frame::cov(*target, correspondences[i]);
135 Eigen::Matrix4d RCR = (target_cov + delta.matrix() * frame::cov(*source, i) * delta.matrix().transpose());
136 RCR(3, 3) = 1.0;
137 mahalanobis[i] = RCR.inverse();
138 mahalanobis[i](3, 3) = 0.0;
139 }
140 };
141
142 if (is_omp_default() || num_threads == 1) {
143#pragma omp parallel for num_threads(num_threads) schedule(guided, 8)
144 for (int i = 0; i < frame::size(*source); i++) {
145 perpoint_task(i);
146 }
147 } else {
148#ifdef GTSAM_POINTS_USE_TBB
149 tbb::parallel_for(tbb::blocked_range<int>(0, frame::size(*source), 8), [&](const tbb::blocked_range<int>& range) {
150 for (int i = range.begin(); i < range.end(); i++) {
151 perpoint_task(i);
152 }
153 });
154#else
155 std::cerr << "error: TBB is not available" << std::endl;
156 abort();
157#endif
158 }
159}

161template <typename TargetFrame, typename SourceFrame, typename IntensityGradients>
162double IntegratedColoredGICPFactor_<TargetFrame, SourceFrame, IntensityGradients>::evaluate(
163 const Eigen::Isometry3d& delta,
164 Eigen::Matrix<double, 6, 6>* H_target,
165 Eigen::Matrix<double, 6, 6>* H_source,
166 Eigen::Matrix<double, 6, 6>* H_target_source,
167 Eigen::Matrix<double, 6, 1>* b_target,
168 Eigen::Matrix<double, 6, 1>* b_source) const {
169 //
170 if (correspondences.size() != frame::size(*source)) {
171 update_correspondences(delta);
172 }
173
174 const double geometric_term_weight = 1.0 - photometric_term_weight;
175
176 const auto perpoint_task = [&](
177 int i,
178 Eigen::Matrix<double, 6, 6>* H_target,
179 Eigen::Matrix<double, 6, 6>* H_source,
180 Eigen::Matrix<double, 6, 6>* H_target_source,
181 Eigen::Matrix<double, 6, 1>* b_target,
182 Eigen::Matrix<double, 6, 1>* b_source) {
183 const long target_index = correspondences[i];
184 if (target_index < 0) {
185 return 0.0;
186 }
187
188 // source atributes
189 const auto& mean_A = frame::point(*source, i);
190 const auto& cov_A = frame::cov(*source, i);
191 const double intensity_A = frame::intensity(*source, i);
192
193 // target attributes
194 const auto& mean_B = frame::point(*target, target_index);
195 const auto& cov_B = frame::cov(*target, target_index);
196 const auto& normal_B = frame::normal(*target, target_index);
197 const auto& gradient_B = frame::intensity_gradient(*target_gradients, target_index);
198 const double intensity_B = frame::intensity(*target, target_index);
199
200 const Eigen::Vector4d transed_A = delta * mean_A;
201
202 // geometric error
203 const Eigen::Vector4d residual_geom = transed_A - mean_B;
204 const double error_geom = residual_geom.transpose() * geometric_term_weight * mahalanobis[i] * residual_geom;
205
206 // photometric error
207 const Eigen::Vector4d projected = transed_A - (transed_A - mean_B).dot(normal_B) * normal_B;
208 const Eigen::Vector4d offset = projected - mean_B;
209 const double residual_photo = intensity_B + gradient_B.dot(offset) - intensity_A;
210
211 const double error_photo = residual_photo * photometric_term_weight * residual_photo;
212
213 if (!H_target) {
214 return error_geom + error_photo;
215 }
216
217 Eigen::Matrix<double, 4, 6> J_transed_target = Eigen::Matrix<double, 4, 6>::Zero();
218 J_transed_target.block<3, 3>(0, 0) = gtsam::SO3::Hat(transed_A.head<3>());
219 J_transed_target.block<3, 3>(0, 3) = -Eigen::Matrix3d::Identity();
220
221 Eigen::Matrix<double, 4, 6> J_transed_source = Eigen::Matrix<double, 4, 6>::Zero();
222 J_transed_source.block<3, 3>(0, 0) = -delta.linear() * gtsam::SO3::Hat(mean_A.template head<3>());
223 J_transed_source.block<3, 3>(0, 3) = delta.linear();
224
225 // geometric error derivatives
226 const auto& J_egeom_target = J_transed_target;
227 const auto& J_egeom_source = J_transed_source;
228
229 *H_target += J_egeom_target.transpose() * geometric_term_weight * mahalanobis[i] * J_egeom_target;
230 *H_source += J_egeom_source.transpose() * geometric_term_weight * mahalanobis[i] * J_egeom_source;
231 *H_target_source += J_egeom_target.transpose() * geometric_term_weight * mahalanobis[i] * J_egeom_source;
232 *b_target += J_egeom_target.transpose() * geometric_term_weight * mahalanobis[i] * residual_geom;
233 *b_source += J_egeom_source.transpose() * geometric_term_weight * mahalanobis[i] * residual_geom;
234
235 // photometric error derivatives
236 Eigen::Matrix<double, 4, 4> J_projected_transed = Eigen::Matrix4d::Identity() - normal_B * normal_B.transpose();
237 J_projected_transed(3, 3) = 0.0;
238 const auto& J_offset_transed = J_projected_transed;
239
240 Eigen::Matrix<double, 1, 4> J_ephoto_offset = gradient_B;
241 Eigen::Matrix<double, 1, 4> J_ephoto_transed = J_ephoto_offset * J_offset_transed;
242
243 Eigen::Matrix<double, 1, 6> J_ephoto_target = J_ephoto_transed * J_transed_target;
244 Eigen::Matrix<double, 1, 6> J_ephoto_source = J_ephoto_transed * J_transed_source;
245
246 *H_target += J_ephoto_target.transpose() * photometric_term_weight * J_ephoto_target;
247 *H_source += J_ephoto_source.transpose() * photometric_term_weight * J_ephoto_source;
248 *H_target_source += J_ephoto_target.transpose() * photometric_term_weight * J_ephoto_source;
249 *b_target += J_ephoto_target.transpose() * photometric_term_weight * residual_photo;
250 *b_source += J_ephoto_source.transpose() * photometric_term_weight * residual_photo;
251
252 return error_geom + error_photo;
253 };
254
255 if (is_omp_default() || num_threads == 1) {
256 return scan_matching_reduce_omp(perpoint_task, frame::size(*source), num_threads, H_target, H_source, H_target_source, b_target, b_source);
257 } else {
258 return scan_matching_reduce_tbb(perpoint_task, frame::size(*source), H_target, H_source, H_target_source, b_target, b_source);
259 }
