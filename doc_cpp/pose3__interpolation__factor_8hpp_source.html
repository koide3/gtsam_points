28public:
29 virtual gtsam::Vector evaluateError(
30 const gtsam::Pose3& xi,
31 const gtsam::Pose3& xj,
32 const gtsam::Pose3& xk,
33 boost::optional<gtsam::Matrix&> H_xi = boost::none,
34 boost::optional<gtsam::Matrix&> H_xj = boost::none,
35 boost::optional<gtsam::Matrix&> H_xk = boost::none) const override {
36 //
37 const auto& Ri = xi.rotation();
38 const auto& Rj = xj.rotation();
39 const auto& Rk = xk.rotation();
40
41 const auto& ti = xi.translation();
42 const auto& tj = xj.translation();
43 const auto& tk = xk.translation();
44
45 if (!H_xi) {
46 const gtsam::Rot3 Rint = gtsam::interpolate(Ri, Rj, t);
47 const gtsam::Rot3 Re = Rk.between(Rint);
48 const gtsam::Vector3 re = gtsam::Rot3::Logmap(Re);
49 const gtsam::Vector3 te = tk - (1.0 - t) * ti - t * tj;
50
51 return (gtsam::Vector6() << re, te).finished();
52 }
53
54 gtsam::Matrix33 H_Rint_Ri, H_Rint_Rj;
55 const gtsam::Rot3 Rint = gtsam::interpolate(Ri, Rj, t, H_Rint_Ri, H_Rint_Rj);
56
57 // H_Re_Rint == Identity
58 gtsam::Matrix33 H_Re_Rk;
59 const gtsam::Rot3 Re = Rk.between(Rint, H_Re_Rk);
60
61 gtsam::Matrix33 H_re_Re;
62 const gtsam::Vector3 re = gtsam::Rot3::Logmap(Re, H_re_Re);
63
64 const gtsam::Vector3 te = tk - (1.0 - t) * ti - t * tj;
65
66 (*H_xi) = gtsam::Matrix6::Zero();
67 (*H_xj) = gtsam::Matrix6::Zero();
68 (*H_xk) = gtsam::Matrix6::Zero();
69
70 // const gtsam::Matrix3 H_re_Rint = H_re_Re * H_Re_Rint;
71 const auto& H_re_Rint = H_re_Re;
72 H_xi->block<3, 3>(0, 0) = H_re_Rint * H_Rint_Ri;
73 H_xj->block<3, 3>(0, 0) = H_re_Rint * H_Rint_Rj;
74 H_xk->block<3, 3>(0, 0) = H_re_Re * H_Re_Rk;
75
76 H_xi->block<3, 3>(3, 3) = -(1.0 - t) * Ri.matrix();
77 H_xj->block<3, 3>(3, 3) = -t * Rj.matrix();
78 H_xk->block<3, 3>(3, 3) = Rk.matrix();
79
80 return (gtsam::Vector6() << re, te).finished();
81 }

83 static gtsam::Pose3 initial_guess(const gtsam::Pose3& xi, const gtsam::Pose3& xj, const double t) {
84 const gtsam::Rot3 Rint = gtsam::interpolate(xi.rotation(), xj.rotation(), t);
85 const gtsam::Vector3 tint = (1.0 - t) * xi.translation() + t * xj.translation();
86 return gtsam::Pose3(Rint, tint);
87 }

89private:
