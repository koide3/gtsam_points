120template <typename TargetFrame, typename SourceFrame>
121void IntegratedGICPFactor_<TargetFrame, SourceFrame>::update_correspondences(const Eigen::Isometry3d& delta) const {
122 linearization_point = delta;
123
124 bool do_update = true;
125 if (correspondences.size() == frame::size(*source) && (correspondence_update_tolerance_trans > 0.0 || correspondence_update_tolerance_rot > 0.0)) {
126 Eigen::Isometry3d diff = delta.inverse() * last_correspondence_point;
127 double diff_rot = Eigen::AngleAxisd(diff.linear()).angle();
128 double diff_trans = diff.translation().norm();
129 if (diff_rot < correspondence_update_tolerance_rot && diff_trans < correspondence_update_tolerance_trans) {
130 do_update = false;
131 }
132 }
133
134 if (do_update) {
135 last_correspondence_point = delta;
136 }
137
138 correspondences.resize(frame::size(*source));
139
140 switch (mahalanobis_cache_mode) {
141 case FusedCovCacheMode::FULL:
142 mahalanobis_full.resize(frame::size(*source));
143 break;
144 case FusedCovCacheMode::COMPACT:
145 mahalanobis_compact.resize(frame::size(*source));
146 break;
147 case FusedCovCacheMode::NONE:
148 break;
149 }
150
151 const auto perpoint_task = [&](int i) {
152 if (do_update) {
153 Eigen::Vector4d pt = delta * frame::point(*source, i);
154
155 size_t k_index = -1;
156 double k_sq_dist = -1;
157 size_t num_found = target_tree->knn_search(pt.data(), 1, &k_index, &k_sq_dist, max_correspondence_distance_sq);
158 correspondences[i] = (num_found && k_sq_dist < max_correspondence_distance_sq) ? k_index : -1;
159 }
160
161 switch (mahalanobis_cache_mode) {
162 case FusedCovCacheMode::FULL:
163 if (correspondences[i] < 0) {
164 mahalanobis_full[i].setZero();
165 } else {
166 const auto& target_cov = frame::cov(*target, correspondences[i]);
167 const Eigen::Matrix4d RCR = (target_cov + delta.matrix() * frame::cov(*source, i) * delta.matrix().transpose());
168 mahalanobis_full[i].setZero();
169 mahalanobis_full[i].topLeftCorner<3, 3>() = RCR.topLeftCorner<3, 3>().inverse();
170 }
171 break;
172 case FusedCovCacheMode::COMPACT:
173 if (correspondences[i] < 0) {
174 mahalanobis_compact[i].setZero();
175 } else {
176 const auto& target_cov = frame::cov(*target, correspondences[i]);
177 const Eigen::Matrix4d RCR = (target_cov + delta.matrix() * frame::cov(*source, i) * delta.matrix().transpose());
178 const Eigen::Matrix3d maha = RCR.topLeftCorner<3, 3>().inverse();
179 mahalanobis_compact[i] = compact_cov(maha);
180 }
181 break;
182 case FusedCovCacheMode::NONE:
183 break;
184 }
185 };
186
187 if (is_omp_default() || num_threads == 1) {
188#pragma omp parallel for num_threads(num_threads) schedule(guided, 8)
189 for (int i = 0; i < frame::size(*source); i++) {
190 perpoint_task(i);
191 }
192 } else {
193#ifdef GTSAM_POINTS_USE_TBB
194 tbb::parallel_for(tbb::blocked_range<int>(0, frame::size(*source), 8), [&](const tbb::blocked_range<int>& range) {
195 for (int i = range.begin(); i < range.end(); i++) {
196 perpoint_task(i);
197 }
198 });
199#else
200 std::cerr << "error: TBB is not available" << std::endl;
201 abort();
202#endif
203 }

206template <typename TargetFrame, typename SourceFrame>
207double IntegratedGICPFactor_<TargetFrame, SourceFrame>::evaluate(
208 const Eigen::Isometry3d& delta,
209 Eigen::Matrix<double, 6, 6>* H_target,
210 Eigen::Matrix<double, 6, 6>* H_source,
211 Eigen::Matrix<double, 6, 6>* H_target_source,
212 Eigen::Matrix<double, 6, 1>* b_target,
213 Eigen::Matrix<double, 6, 1>* b_source) const {
214 //
215 if (correspondences.size() != frame::size(*source)) {
216 update_correspondences(delta);
217 }
218
219 const auto perpoint_task = [&](
220 int i,
221 Eigen::Matrix<double, 6, 6>* H_target,
222 Eigen::Matrix<double, 6, 6>* H_source,
223 Eigen::Matrix<double, 6, 6>* H_target_source,
224 Eigen::Matrix<double, 6, 1>* b_target,
225 Eigen::Matrix<double, 6, 1>* b_source) {
226 const long target_index = correspondences[i];
227 if (target_index < 0) {
228 return 0.0;
229 }
230
231 const auto& mean_A = frame::point(*source, i);
232 const auto& cov_A = frame::cov(*source, i);
233 const auto& mean_B = frame::point(*target, target_index);
234 const auto& cov_B = frame::cov(*target, target_index);
235
236 const Eigen::Vector4d transed_mean_A = delta * mean_A;
237 const Eigen::Vector4d residual = mean_B - transed_mean_A;
238
239 Eigen::Matrix4d mahalanobis;
240 switch (mahalanobis_cache_mode) {
241 case FusedCovCacheMode::FULL:
242 mahalanobis = mahalanobis_full[i];
243 break;
244 case FusedCovCacheMode::COMPACT:
245 mahalanobis = uncompact_cov(mahalanobis_compact[i]);
246 break;
247 case FusedCovCacheMode::NONE: {
248 const auto& delta_l = linearization_point; // Delta at the linearization point
249 const Eigen::Matrix4d RCR = (cov_B + delta_l.matrix() * cov_A * delta_l.matrix().transpose());
250 mahalanobis.setZero();
251 mahalanobis.topLeftCorner<3, 3>() = RCR.topLeftCorner<3, 3>().inverse();
252 } break;
253 }
254
255 const double error = residual.transpose() * mahalanobis * residual;
256 if (H_target == nullptr) {
257 return error;
258 }
259
260 Eigen::Matrix<double, 4, 6> J_target = Eigen::Matrix<double, 4, 6>::Zero();
261 J_target.block<3, 3>(0, 0) = -gtsam::SO3::Hat(transed_mean_A.head<3>());
262 J_target.block<3, 3>(0, 3) = Eigen::Matrix3d::Identity();
263
264 Eigen::Matrix<double, 4, 6> J_source = Eigen::Matrix<double, 4, 6>::Zero();
265 J_source.block<3, 3>(0, 0) = delta.linear() * gtsam::SO3::Hat(mean_A.template head<3>());
266 J_source.block<3, 3>(0, 3) = -delta.linear();
267
268 Eigen::Matrix<double, 6, 4> J_target_mahalanobis = J_target.transpose() * mahalanobis;
269 Eigen::Matrix<double, 6, 4> J_source_mahalanobis = J_source.transpose() * mahalanobis;
270
271 *H_target += J_target_mahalanobis * J_target;
272 *H_source += J_source_mahalanobis * J_source;
273 *H_target_source += J_target_mahalanobis * J_source;
274 *b_target += J_target_mahalanobis * residual;
275 *b_source += J_source_mahalanobis * residual;
276
277 return error;
278 };
279
280 if (is_omp_default() || num_threads == 1) {
281 return scan_matching_reduce_omp(perpoint_task, frame::size(*source), num_threads, H_target, H_source, H_target_source, b_target, b_source);
282 } else {
283 return scan_matching_reduce_tbb(perpoint_task, frame::size(*source), H_target, H_source, H_target_source, b_target, b_source);
284 }
