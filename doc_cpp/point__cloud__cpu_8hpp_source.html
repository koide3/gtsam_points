160template <typename Func>
161PointCloudCPU::Ptr filter(const PointCloud::ConstPtr& points, const Func& pred) {
162 std::vector<int> indices;
163 indices.reserve(points->size());
164 for (int i = 0; i < points->size(); i++) {
165 if (pred(points->points[i])) {
166 indices.push_back(i);
167 }
168 }
169 return sample(points, indices);
170}

194template <typename Compare>

196 std::vector<int> indices(points->size());
