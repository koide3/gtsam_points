81 template <typename TargetFrame, typename SourceFrame, typename IntensityGradients>
82 void IntegratedColorConsistencyFactor_<TargetFrame, SourceFrame, IntensityGradients>::update_correspondences(const Eigen::Isometry3d& delta) const {
83  bool do_update = true;
84  if (correspondences.size() == frame::size(*source) && (correspondence_update_tolerance_trans > 0.0 || correspondence_update_tolerance_rot > 0.0)) {
85  Eigen::Isometry3d diff = delta.inverse() * last_correspondence_point;
86  double diff_rot = Eigen::AngleAxisd(diff.linear()).angle();
87  double diff_trans = diff.translation().norm();
88  if (diff_rot < correspondence_update_tolerance_rot && diff_trans < correspondence_update_tolerance_trans) {
89  do_update = false;
90  }
91  }
92 
93  if (!do_update) {
94  return;
95  }
96 
97  last_correspondence_point = delta;
98  correspondences.resize(frame::size(*source));
99 
100  const auto perpoint_task = [&](int i) {
101  Eigen::Vector4d pt = delta * frame::point(*source, i);
102  pt[3] = frame::intensity(*source, i);
103 
104  size_t k_index = -1;
105  double k_sq_dist = -1;
106 
107  size_t num_found = target_tree->knn_search(pt.data(), 1, &k_index, &k_sq_dist, max_correspondence_distance_sq);
108  correspondences[i] = (num_found && k_sq_dist < max_correspondence_distance_sq) ? k_index : -1;
109  };
110 
111  if (is_omp_default() || num_threads == 1) {
112 #pragma omp parallel for num_threads(num_threads) schedule(guided, 8)
113  for (int i = 0; i < frame::size(*source); i++) {
114  perpoint_task(i);
115  }
116  } else {
117 #ifdef GTSAM_POINTS_USE_TBB
118  tbb::parallel_for(tbb::blocked_range<int>(0, frame::size(*source), 8), [&](const tbb::blocked_range<int>& range) {
119  for (int i = range.begin(); i < range.end(); i++) {
120  perpoint_task(i);
121  }
122  });
123 #else
124  std::cerr << "error: TBB is not available" << std::endl;
125  abort();
126 #endif
127  }
128 }

130 template <typename TargetFrame, typename SourceFrame, typename IntensityGradients>
131 double IntegratedColorConsistencyFactor_<TargetFrame, SourceFrame, IntensityGradients>::evaluate(
132  const Eigen::Isometry3d& delta,
133  Eigen::Matrix<double, 6, 6>* H_target,
134  Eigen::Matrix<double, 6, 6>* H_source,
135  Eigen::Matrix<double, 6, 6>* H_target_source,
136  Eigen::Matrix<double, 6, 1>* b_target,
137  Eigen::Matrix<double, 6, 1>* b_source) const {
138  //
139  if (correspondences.size() != frame::size(*source)) {
140  update_correspondences(delta);
141  }
142 
143  const auto perpoint_task = [&](
144  int i,
145  Eigen::Matrix<double, 6, 6>* H_target,
146  Eigen::Matrix<double, 6, 6>* H_source,
147  Eigen::Matrix<double, 6, 6>* H_target_source,
148  Eigen::Matrix<double, 6, 1>* b_target,
149  Eigen::Matrix<double, 6, 1>* b_source) {
150  const long target_index = correspondences[i];
151  if (target_index < 0) {
152  return 0.0;
153  }
154 
155  // source atributes
156  const auto& mean_A = frame::point(*source, i);
157  const double intensity_A = frame::intensity(*source, i);
158 
159  // target attributes
160  const auto& mean_B = frame::point(*target, target_index);
161  const auto& normal_B = frame::normal(*target, target_index);
162  const auto& gradient_B = frame::intensity_gradient(*target_gradients, target_index);
163  const double intensity_B = frame::intensity(*target, target_index);
164 
165  const Eigen::Vector4d transed_A = delta * mean_A;
166 
167  // photometric error
168  const Eigen::Vector4d projected = transed_A - (transed_A - mean_B).dot(normal_B) * normal_B;
169  const Eigen::Vector4d offset = projected - mean_B;
170  const double error_photo = intensity_B + gradient_B.dot(offset) - intensity_A;
171 
172  const double err = 0.5 * error_photo * photometric_term_weight * error_photo;
173 
174  if (!H_target) {
175  return err;
176  }
177 
178  Eigen::Matrix<double, 4, 6> J_transed_target = Eigen::Matrix<double, 4, 6>::Zero();
179  J_transed_target.block<3, 3>(0, 0) = gtsam::SO3::Hat(transed_A.head<3>());
180  J_transed_target.block<3, 3>(0, 3) = -Eigen::Matrix3d::Identity();
181 
182  Eigen::Matrix<double, 4, 6> J_transed_source = Eigen::Matrix<double, 4, 6>::Zero();
183  J_transed_source.block<3, 3>(0, 0) = -delta.linear() * gtsam::SO3::Hat(mean_A.template head<3>());
184  J_transed_source.block<3, 3>(0, 3) = delta.linear();
185 
186  // photometric error derivatives
187  Eigen::Matrix<double, 4, 4> J_projected_transed = Eigen::Matrix4d::Identity() - normal_B * normal_B.transpose();
188  J_projected_transed(3, 3) = 0.0;
189  const auto& J_offset_transed = J_projected_transed;
190 
191  Eigen::Matrix<double, 1, 4> J_ephoto_offset = gradient_B;
192  Eigen::Matrix<double, 1, 4> J_ephoto_transed = J_ephoto_offset * J_offset_transed;
193 
194  Eigen::Matrix<double, 1, 6> J_ephoto_target = J_ephoto_transed * J_transed_target;
195  Eigen::Matrix<double, 1, 6> J_ephoto_source = J_ephoto_transed * J_transed_source;
196 
197  *H_target += J_ephoto_target.transpose() * photometric_term_weight * J_ephoto_target;
198  *H_source += J_ephoto_source.transpose() * photometric_term_weight * J_ephoto_source;
199  *H_target_source += J_ephoto_target.transpose() * photometric_term_weight * J_ephoto_source;
200  *b_target += J_ephoto_target.transpose() * photometric_term_weight * error_photo;
201  *b_source += J_ephoto_source.transpose() * photometric_term_weight * error_photo;
202 
203  return err;
204  };
205 
206  double sum_errors_photo = 0.0;
207 
208  std::vector<Eigen::Matrix<double, 6, 6>> Hs_target;
209  std::vector<Eigen::Matrix<double, 6, 6>> Hs_source;
210  std::vector<Eigen::Matrix<double, 6, 6>> Hs_target_source;
211  std::vector<Eigen::Matrix<double, 6, 1>> bs_target;
212  std::vector<Eigen::Matrix<double, 6, 1>> bs_source;
213 
214  if (H_target && H_source && H_target_source && b_target && b_source) {
215  Hs_target.resize(num_threads, Eigen::Matrix<double, 6, 6>::Zero());
216  Hs_source.resize(num_threads, Eigen::Matrix<double, 6, 6>::Zero());
217  Hs_target_source.resize(num_threads, Eigen::Matrix<double, 6, 6>::Zero());
218  bs_target.resize(num_threads, Eigen::Matrix<double, 6, 1>::Zero());
219  bs_source.resize(num_threads, Eigen::Matrix<double, 6, 1>::Zero());
220  }
221 
222  if (is_omp_default() || num_threads == 1) {
223  return scan_matching_reduce_omp(perpoint_task, frame::size(*source), num_threads, H_target, H_source, H_target_source, b_target, b_source);
224  } else {
225  return scan_matching_reduce_tbb(perpoint_task, frame::size(*source), H_target, H_source, H_target_source, b_target, b_source);
226  }
