58template <typename PointCloud>
59MinCutResult min_cut_(const PointCloud& points, const NearestNeighborSearch& search, const size_t source_pt_index, const MinCutParams& params) {
60 // Find k-nearest neighbors for all points
61 std::vector<size_t> all_neighbors(frame::size(points) * params.k_neighbors);
62 std::vector<double> all_neighbor_sq_dists(frame::size(points) * params.k_neighbors);
63#pragma omp parallel for num_threads(params.num_threads) schedule(guided, 4)
64 for (size_t i = 0; i < frame::size(points); i++) {
65 size_t* k_neighbors = all_neighbors.data() + i * params.k_neighbors;
66 double* k_sq_dists = all_neighbor_sq_dists.data() + i * params.k_neighbors;
67 search.knn_search(frame::point(points, i).data(), params.k_neighbors, k_neighbors, k_sq_dists);
68 }
69
70 // clang-format off
71 using Traits = boost::adjacency_list_traits<boost::vecS, boost::vecS, boost::directedS>;
72 using Graph = boost::adjacency_list<boost::vecS, boost::vecS, boost::directedS,
73 boost::property<boost::vertex_index_t, long,//
74 boost::property<boost::vertex_color_t, boost::default_color_type,//
75 boost::property<boost::vertex_distance_t, long,//
76 boost::property<boost::vertex_predecessor_t, Traits::edge_descriptor>>>>, //
77 boost::property<boost::edge_capacity_t, double,//
78 boost::property<boost::edge_residual_capacity_t, double,//
79 boost::property<boost::edge_reverse_t, Traits::edge_descriptor>>>>;
80 using EdgeDescriptor = Traits::edge_descriptor;
81 // clang-format on
82
83 // Graph construction
84 Graph g(frame::size(points));
85 boost::property_map<Graph, boost::edge_capacity_t>::type capacity = get(boost::edge_capacity, g);
86 boost::property_map<Graph, boost::edge_residual_capacity_t>::type residual_capacity = get(boost::edge_residual_capacity, g);
87 boost::property_map<Graph, boost::edge_reverse_t>::type reverse_edges = get(boost::edge_reverse, g);
88 boost::property_map<Graph, boost::vertex_color_t>::type color = get(boost::vertex_color, g);
89
90 const auto add_edge = [&](size_t source, size_t target, double weight) {
91 const auto edge = boost::add_edge(source, target, g);
92 const auto reverse_edge = boost::add_edge(target, source, g);
93 if (!edge.second || !reverse_edge.second) {
94 return;
95 }
96
97 capacity[edge.first] = weight;
98 capacity[reverse_edge.first] = 0.0;
99 reverse_edges[edge.first] = reverse_edge.first;
100 reverse_edges[reverse_edge.first] = edge.first;
101 };
102
103 const double inv_dist_sq_sigma = 1.0 / (params.distance_sigma * params.distance_sigma);
104 const double inv_angle_sq_sigma = 1.0 / (params.angle_sigma * params.angle_sigma);
105
106 std::vector<size_t> sink_pt_indices;
107 for (size_t i = 0; i < frame::size(points); i++) {
108 const size_t* neighbors = all_neighbors.data() + i * params.k_neighbors;
109 const double* neighbor_sq_dists = all_neighbor_sq_dists.data() + i * params.k_neighbors;
110
111 const Eigen::Vector4d& normal = frame::normal(points, i);
112
113 const double dist_from_source = (frame::point(points, i) - frame::point(points, source_pt_index)).norm();
114
115 // Mark as foreground if the point is within the foreground mask radius
116 if (i != source_pt_index && dist_from_source < params.foreground_mask_radius) {
117 add_edge(i, source_pt_index, params.foreground_weight);
118 }
119 // Mark as background if the point is outside the background mask radius
120 if (dist_from_source > params.background_mask_radius) {
121 sink_pt_indices.emplace_back(i);
122 }
123
124 // Add connectivity edges to the neighbors
125 for (size_t k = 0; k < params.k_neighbors; k++) {
126 const double weight_dist = std::exp(-neighbor_sq_dists[k] * inv_dist_sq_sigma);
127 const double normal_diff = std::acos(std::abs(normal.dot(frame::normal(points, neighbors[k]))));
128 const double weight_angle = std::exp(-normal_diff * normal_diff * inv_angle_sq_sigma);
129 add_edge(i, neighbors[k], weight_dist * weight_angle);
130 }
131 }
132
133 if (sink_pt_indices.empty()) {
134 std::cerr << "No sink points" << std::endl;
135 return MinCutResult();
136 }
137
138 size_t sink_pt_index = sink_pt_indices.front();
139 for (size_t i = 1; i < sink_pt_indices.size(); i++) {
140 add_edge(sink_pt_indices[i], sink_pt_index, params.background_weight);
141 }
142
143 // Run the min-cut algorithm
144 MinCutResult result;
145 result.source_index = source_pt_index;
146 result.sink_index = sink_pt_index;
147 result.max_flow = boost::boykov_kolmogorov_max_flow(g, source_pt_index, sink_pt_index);
148
149 // Extract the cluster
150 const auto source_color = color[source_pt_index];
151 for (size_t i = 0; i < frame::size(points); i++) {
152 if (color[i] == source_color) {
153 result.cluster_indices.emplace_back(i);
154 }
155 }
156
157 return result;
158}

160template <typename PointCloud>
161MinCutResult min_cut_(const PointCloud& points, const NearestNeighborSearch& search, const Eigen::Vector4d& source_pt, const MinCutParams& params) {
162 size_t source_pt_index;
163 double sq_dist;
164 if (!search.knn_search(source_pt.data(), 1, &source_pt_index, &sq_dist)) {
165 std::cerr << "Failed to find the source point" << std::endl;
166 return MinCutResult();
167 }
168
169 if (sq_dist > params.foreground_mask_radius * params.foreground_mask_radius) {
170 std::cerr << "warning: The source point is too far from the point cloud" << std::endl;
171 }
172
173 return min_cut_(points, search, source_pt_index, params);
174}
