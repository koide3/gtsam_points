31  const gtsam::Pose3 delta = xi.between(xj); // (xi^-1 * xj)
32  const gtsam::Pose3 error = delta.between(Tij); // (xj_inv * xi) * Tij;
33  return gtsam::Pose3::Logmap(error);
34  }
35 
36  gtsam::Matrix6 H_delta_i;
37  gtsam::Matrix6 H_delta_j;
38  gtsam::Pose3 delta = xi.between(xj, H_delta_i, H_delta_j);
39 
40  gtsam::Matrix6 H_error_delta;
41  gtsam::Matrix6 H_error_Tij;
42  gtsam::Pose3 error = delta.between(Tij, H_error_delta, H_error_Tij);
43 
44  gtsam::Matrix6 H_log_error;
45  gtsam::Vector6 log = gtsam::Pose3::Logmap(error, H_log_error);
46 
47  (*H_xi) = H_log_error * H_error_delta * H_delta_i;
48  (*H_xj) = H_log_error * H_error_delta * H_delta_j;
49  (*H_Tij) = H_log_error * H_error_Tij;
50 
51  return log;
52  }
