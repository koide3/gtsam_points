49 template <typename TargetFrame, typename SourceFrame>
50 double IntegratedCT_GICPFactor_<TargetFrame, SourceFrame>::error(const gtsam::Values& values) const {
51  this->update_poses(values);
52  if (this->correspondences.size() != frame::size(*this->source)) {
53  this->update_correspondences();
54  }
55 
56  const auto perpoint_task = [&](
57  int i,
58  Eigen::Matrix<double, 6, 6>* H_target,
59  Eigen::Matrix<double, 6, 6>* H_source,
60  Eigen::Matrix<double, 6, 6>* H_target_source,
61  Eigen::Matrix<double, 6, 1>* b_target,
62  Eigen::Matrix<double, 6, 1>* b_source) {
63  const long target_index = this->correspondences[i];
64  if (target_index < 0) {
65  return 0.0;
66  }
67 
68  const int time_index = this->time_indices[i];
69  const Eigen::Isometry3d pose(this->source_poses[time_index].matrix());
70 
71  const auto& source_pt = frame::point(*this->source, i);
72  const auto& target_pt = frame::point(*this->target, target_index);
73 
74  const Eigen::Vector4d transed_source_pt = pose * source_pt;
75  const Eigen::Vector4d residual = transed_source_pt - target_pt;
76  const double error = 0.5 * residual.transpose() * mahalanobis[i] * residual;
77 
78  return error;
79  };
80 
81  if (is_omp_default() || this->num_threads == 1) {
82  return scan_matching_reduce_omp(perpoint_task, frame::size(*this->source), this->num_threads, nullptr, nullptr, nullptr, nullptr, nullptr);
83  } else {
84  return scan_matching_reduce_tbb(perpoint_task, frame::size(*this->source), nullptr, nullptr, nullptr, nullptr, nullptr);
85  }
86 }

88 template <typename TargetFrame, typename SourceFrame>
89 boost::shared_ptr<gtsam::GaussianFactor> IntegratedCT_GICPFactor_<TargetFrame, SourceFrame>::linearize(const gtsam::Values& values) const {
90  this->update_poses(values);
91  this->update_correspondences();
92 
93  const auto perpoint_task = [&](
94  int i,
95  Eigen::Matrix<double, 6, 6>* H_00,
96  Eigen::Matrix<double, 6, 6>* H_11,
97  Eigen::Matrix<double, 6, 6>* H_01,
98  Eigen::Matrix<double, 6, 1>* b_0,
99  Eigen::Matrix<double, 6, 1>* b_1) {
100  const long target_index = this->correspondences[i];
101  if (target_index < 0) {
102  return 0.0;
103  }
104 
105  const int time_index = this->time_indices[i];
106 
107  const Eigen::Isometry3d pose(this->source_poses[time_index].matrix());
108  const auto& H_pose_0 = this->pose_derivatives_t0[time_index];
109  const auto& H_pose_1 = this->pose_derivatives_t1[time_index];
110 
111  const auto& source_pt = frame::point(*this->source, i);
112  const auto& target_pt = frame::point(*this->target, target_index);
113 
114  gtsam::Matrix46 H_transed_pose = gtsam::Matrix46::Zero();
115  H_transed_pose.block<3, 3>(0, 0) = pose.linear() * -gtsam::SO3::Hat(source_pt.template head<3>());
116  H_transed_pose.block<3, 3>(0, 3) = pose.linear();
117  const Eigen::Vector4d transed_source_pt = pose * source_pt;
118 
119  const auto& H_residual_pose = H_transed_pose;
120  const Eigen::Vector4d residual = transed_source_pt - target_pt;
121 
122  const gtsam::Matrix46 H_0 = H_residual_pose * H_pose_0;
123  const gtsam::Matrix46 H_1 = H_residual_pose * H_pose_1;
124 
125  const gtsam::Vector4 mahalanobis_residual = mahalanobis[i] * residual;
126  const gtsam::Matrix64 H_0_mahalanobis = H_0.transpose() * mahalanobis[i];
127  const gtsam::Matrix64 H_1_mahalanobis = H_1.transpose() * mahalanobis[i];
128 
129  const double error = 0.5 * residual.transpose() * mahalanobis_residual;
130  *H_00 += H_0_mahalanobis * H_0;
131  *H_11 += H_1_mahalanobis * H_1;
132  *H_01 += H_0_mahalanobis * H_1;
133  *b_0 += H_0.transpose() * mahalanobis_residual;
134  *b_1 += H_1.transpose() * mahalanobis_residual;
135 
136  return error;
137  };
138 
139  double error = 0.0;
140  gtsam::Matrix6 H_00;
141  gtsam::Matrix6 H_01;
142  gtsam::Matrix6 H_11;
143  gtsam::Vector6 b_0;
144  gtsam::Vector6 b_1;
145 
146  if (is_omp_default() || this->num_threads == 1) {
147  error = scan_matching_reduce_omp(perpoint_task, frame::size(*this->source), this->num_threads, &H_00, &H_11, &H_01, &b_0, &b_1);
148  } else {
149  error = scan_matching_reduce_tbb(perpoint_task, frame::size(*this->source), &H_00, &H_11, &H_01, &b_0, &b_1);
150  }
151 
152  auto factor = gtsam::HessianFactor::shared_ptr(new gtsam::HessianFactor(this->keys_[0], this->keys_[1], H_00, H_01, -b_0, H_11, -b_1, error));
153  return factor;
154 }

156 template <typename TargetFrame, typename SourceFrame>
157 void IntegratedCT_GICPFactor_<TargetFrame, SourceFrame>::update_correspondences() const {
158  this->correspondences.resize(frame::size(*this->source));
159  this->mahalanobis.resize(frame::size(*this->source));
160 
161  const auto perpoint_task = [&](int i) {
162  const int time_index = this->time_indices[i];
163  const Eigen::Matrix4d pose = this->source_poses[time_index].matrix();
164 
165  const auto& pt = frame::point(*this->source, i);
166  const Eigen::Vector4d transed_pt = pose * pt;
167 
168  size_t k_index = -1;
169  double k_sq_dist = std::numeric_limits<double>::max();
170  size_t num_found = this->target_tree->knn_search(transed_pt.data(), 1, &k_index, &k_sq_dist, this->max_correspondence_distance_sq);
171 
172  if (num_found == 0 || k_sq_dist > this->max_correspondence_distance_sq) {
173  this->correspondences[i] = -1;
174  this->mahalanobis[i].setZero();
175  } else {
176  this->correspondences[i] = k_index;
177 
178  const long target_index = this->correspondences[i];
179  const auto& cov_A = frame::cov(*this->source, i);
180  const auto& cov_B = frame::cov(*this->target, target_index);
181  const Eigen::Matrix4d RCR = (cov_B + pose * cov_A * pose.transpose());
182 
183  mahalanobis[i].setZero();
184  mahalanobis[i].block<3, 3>(0, 0) = RCR.block<3, 3>(0, 0).inverse();
185  }
186  };
187 
188  if (is_omp_default() || this->num_threads == 1) {
189 #pragma omp parallel for num_threads(this->num_threads) schedule(guided, 8)
190  for (int i = 0; i < frame::size(*this->source); i++) {
191  perpoint_task(i);
192  }
193  } else {
194 #ifdef GTSAM_POINTS_USE_TBB
195  tbb::parallel_for(tbb::blocked_range<int>(0, frame::size(*this->source), 8), [&](const tbb::blocked_range<int>& range) {
196  for (int i = range.begin(); i < range.end(); i++) {
197  perpoint_task(i);
198  }
199  });
200 #else
201  std::cerr << "error: TBB is not available" << std::endl;
202  abort();
203 #endif
204  }
205 }
