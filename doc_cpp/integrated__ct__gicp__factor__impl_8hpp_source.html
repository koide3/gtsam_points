42 template <typename TargetFrame, typename SourceFrame>
43 double IntegratedCT_GICPFactor_<TargetFrame, SourceFrame>::error(const gtsam::Values& values) const {
44  this->update_poses(values);
45  if (this->correspondences.size() != frame::size(*this->source)) {
46  this->update_correspondences();
47  }
48 
49  double sum_errors = 0.0;
50 #pragma omp parallel for reduction(+ : sum_errors) schedule(guided, 8) num_threads(this->num_threads)
51  for (int i = 0; i < frame::size(*this->source); i++) {
52  const long target_index = this->correspondences[i];
53  if (target_index < 0) {
54  continue;
55  }
56 
57  const int time_index = this->time_indices[i];
58  const Eigen::Isometry3d pose(this->source_poses[time_index].matrix());
59 
60  const auto& source_pt = frame::point(*this->source, i);
61  const auto& target_pt = frame::point(*this->target, target_index);
62 
63  Eigen::Vector4d transed_source_pt = pose * source_pt;
64  Eigen::Vector4d error = transed_source_pt - target_pt;
65 
66  sum_errors += 0.5 * error.transpose() * mahalanobis[i] * error;
67  }
68 
69  return sum_errors;
70 }

72 template <typename TargetFrame, typename SourceFrame>
73 boost::shared_ptr<gtsam::GaussianFactor> IntegratedCT_GICPFactor_<TargetFrame, SourceFrame>::linearize(const gtsam::Values& values) const {
74  this->update_poses(values);
75  this->update_correspondences();
76 
77  double sum_errors = 0.0;
78  std::vector<gtsam::Matrix6> Hs_00(this->num_threads, gtsam::Matrix6::Zero());
79  std::vector<gtsam::Matrix6> Hs_01(this->num_threads, gtsam::Matrix6::Zero());
80  std::vector<gtsam::Matrix6> Hs_11(this->num_threads, gtsam::Matrix6::Zero());
81  std::vector<gtsam::Vector6> bs_0(this->num_threads, gtsam::Vector6::Zero());
82  std::vector<gtsam::Vector6> bs_1(this->num_threads, gtsam::Vector6::Zero());
83 
84  gtsam::Matrix6 H_00 = gtsam::Matrix6::Zero();
85  gtsam::Matrix6 H_01 = gtsam::Matrix6::Zero();
86  gtsam::Matrix6 H_11 = gtsam::Matrix6::Zero();
87  gtsam::Vector6 b_0 = gtsam::Vector6::Zero();
88  gtsam::Vector6 b_1 = gtsam::Vector6::Zero();
89 
90 #pragma omp parallel for reduction(+ : sum_errors) schedule(guided, 8) num_threads(this->num_threads)
91  for (int i = 0; i < frame::size(*this->source); i++) {
92  const long target_index = this->correspondences[i];
93  if (target_index < 0) {
94  continue;
95  }
96 
97  const int time_index = this->time_indices[i];
98 
99  const Eigen::Isometry3d pose(this->source_poses[time_index].matrix());
100  const auto& H_pose_0 = this->pose_derivatives_t0[time_index];
101  const auto& H_pose_1 = this->pose_derivatives_t1[time_index];
102 
103  const auto& source_pt = frame::point(*this->source, i);
104  const auto& target_pt = frame::point(*this->target, target_index);
105 
106  gtsam::Matrix46 H_transed_pose = gtsam::Matrix46::Zero();
107  H_transed_pose.block<3, 3>(0, 0) = pose.linear() * -gtsam::SO3::Hat(source_pt.template head<3>());
108  H_transed_pose.block<3, 3>(0, 3) = pose.linear();
109  const Eigen::Vector4d transed_source_pt = pose * source_pt;
110 
111  const auto& H_error_pose = H_transed_pose;
112  const Eigen::Vector4d error = transed_source_pt - target_pt;
113 
114  const gtsam::Matrix46 H_0 = H_error_pose * H_pose_0;
115  const gtsam::Matrix46 H_1 = H_error_pose * H_pose_1;
116 
117  int thread_num = 0;
118 #ifdef _OPENMP
119  thread_num = omp_get_thread_num();
120 #endif
121 
122  const gtsam::Vector4 mahalanobis_error = mahalanobis[i] * error;
123  const gtsam::Matrix64 H_0_mahalanobis = H_0.transpose() * mahalanobis[i];
124  const gtsam::Matrix64 H_1_mahalanobis = H_1.transpose() * mahalanobis[i];
125 
126  sum_errors += 0.5 * error.transpose() * mahalanobis_error;
127  Hs_00[thread_num] += H_0_mahalanobis * H_0;
128  Hs_11[thread_num] += H_1_mahalanobis * H_1;
129  Hs_01[thread_num] += H_0_mahalanobis * H_1;
130  bs_0[thread_num] += H_0.transpose() * mahalanobis_error;
131  bs_1[thread_num] += H_1.transpose() * mahalanobis_error;
132  }
133 
134  for (int i = 1; i < Hs_00.size(); i++) {
135  Hs_00[0] += Hs_00[i];
136  Hs_11[0] += Hs_11[i];
137  Hs_01[0] += Hs_01[i];
138  bs_0[0] += bs_0[i];
139  bs_1[0] += bs_1[i];
140  }
141 
142  auto factor = gtsam::HessianFactor::shared_ptr(
143  new gtsam::HessianFactor(this->keys_[0], this->keys_[1], Hs_00[0], Hs_01[0], -bs_0[0], Hs_11[0], -bs_1[0], sum_errors));
144  return factor;
145 }

147 template <typename TargetFrame, typename SourceFrame>
148 void IntegratedCT_GICPFactor_<TargetFrame, SourceFrame>::update_correspondences() const {
149  this->correspondences.resize(frame::size(*this->source));
150  this->mahalanobis.resize(frame::size(*this->source));
151 
152 #pragma omp parallel for schedule(guided, 8) num_threads(this->num_threads)
153  for (int i = 0; i < frame::size(*this->source); i++) {
154  const int time_index = this->time_indices[i];
155  const Eigen::Matrix4d pose = this->source_poses[time_index].matrix();
156 
157  const auto& pt = frame::point(*this->source, i);
158  const Eigen::Vector4d transed_pt = pose * pt;
159 
160  size_t k_index = -1;
161  double k_sq_dist = std::numeric_limits<double>::max();
162  size_t num_found = this->target_tree->knn_search(transed_pt.data(), 1, &k_index, &k_sq_dist);
163 
164  if (num_found == 0 || k_sq_dist > this->max_correspondence_distance_sq) {
165  this->correspondences[i] = -1;
166  this->mahalanobis[i].setZero();
167  } else {
168  this->correspondences[i] = k_index;
169 
170  const long target_index = this->correspondences[i];
171  const auto& cov_A = frame::cov(*this->source, i);
172  const auto& cov_B = frame::cov(*this->target, target_index);
173  const Eigen::Matrix4d RCR = (cov_B + pose * cov_A * pose.transpose());
174 
175  mahalanobis[i].setZero();
176  mahalanobis[i].block<3, 3>(0, 0) = RCR.block<3, 3>(0, 0).inverse();
177  }
178  }
